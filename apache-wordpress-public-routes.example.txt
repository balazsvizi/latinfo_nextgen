# WordPress + Latinfo publikus útvonalak (public_html/.htaccess)
#
# Illeszd be a WordPress „BEGIN WordPress” sor ELÉ.
# (Ha már van RewriteEngine On, ne ismételd meg.)
#
# FONTOS: a megjelenit.php → /event/ átirányítás CSAK THE_REQUEST-tel!
# Enélkül /event/ ↔ megjelenit.php végtelen redirect (ERR_TOO_MANY_REDIRECTS).

# --- Latinfo publikus útvonalak (kezdet) ---

RewriteEngine On

# Egy esemény: /event/{slug}/  →  nextgen/events/megjelenit.php
RewriteRule ^event/([^/]+)/?$ nextgen/events/megjelenit.php?slug=$1 [L,QSA,NC]

# Naptár főoldal: /events/
RewriteRule ^events/?$ nextgen/events/index.php [L,QSA,NC]

# Partnerportál (tartalék)
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^partnerportal/?$ nextgen/partner/login.php [L,QSA,NC]

RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^partnerportal/(.+)$ nextgen/partner/$1 [L,QSA,NC]

# Régi URL-ek
RewriteRule ^partners/?(.*)$ /partnerportal/$1 [R=301,L,QSA,NC]
RewriteRule ^nextgen/partner/login\.php$ /partnerportal/ [R=301,L,QSA,NC]
RewriteRule ^nextgen/events/?$ /events/ [R=301,L,QSA,NC]
RewriteRule ^esemenyek/([^/]+)/?$ /event/$1/ [R=301,L,NC]
RewriteRule ^esemenyek/?$ /events/ [R=301,L,QSA,NC]

# Régi közvetlen megjelenit.php hívás → szép URL
# THE_REQUEST = a böngésző eredeti kérése (belső rewrite után NEM illeszkedik)
RewriteCond %{THE_REQUEST} \s/+nextgen/events/megjelenit\.php[\s?] [NC]
RewriteCond %{QUERY_STRING} (^|&)slug=([^&]+)
RewriteRule ^nextgen/events/megjelenit\.php$ /event/%2/? [R=301,L,NE,QSD]

# --- Latinfo publikus útvonalak (vég) ---

# BEGIN WordPress
# ... (a meglévő WordPress blokk változatlanul)
